PurposeLimitationCompliance(X, Y) -< DataSubject(Y), DataController(X), ProcessingData(X, Y).
~PurposeLimitationCompliance(X, Y) -<  ~MinimizedData(X, Y).

%Transparent
~PurposeLimitationCompliance(X, Y) <- ~SpecificProcessingContext(X, Y).

~PurposeLimitationCompliance(X, Y) <- ~MinimalDataCollection(X, Y).
MinimalDataCollection(X, Y) -< DataMinimization(X, Y).
DataMinimization(X, Y) <- MinimizedData(X, Y).

PurposeLimitationCompliance(X, Y) <- ~DataRetentionPeriod(X, Y).
DataRetentionPeriod(X, Y) <- RetentionPeriodDefined(X, Y, "Defined").

~PurposeLimitationCompliance(X, Y) <- ~ExplicitPurpose(X, Y).
ExplicitPurpose(X, Y) <- PurposeType(Y, X, "Explicit").
