HumanitarianEmergency(Y) <- IncapableOfGivingConsent(Y, X).

~VitalInterestCompliance(X, Y) -< ~ExplicitConsent(Y, X).
ExplicitConsent(Y, X) <- ~IncapableOfGivingConsent(Y, X).
